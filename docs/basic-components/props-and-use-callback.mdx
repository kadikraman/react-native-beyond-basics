# Props and useCallback

```tsx title="src/components/MoodPicker.tsx"
import React from "react";
import { Pressable, StyleSheet, Text, View } from "react-native";

const moodOptions = [
  { emoji: "ðŸ˜„", description: "amazing" },
  { emoji: "ðŸ˜Š", description: "great" },
  { emoji: "ðŸ™‚", description: "alright" },
  { emoji: "ðŸ™", description: "sad" },
  { emoji: "ðŸ˜”", description: "miserable" },
];

type MoodOptionType = typeof moodOptions[number];

type MoodPickerProps = {
  onAddMood: (newMood: MoodOptionType) => void;
};

export const MoodPicker: React.FC<MoodPickerProps> = ({ onAddMood }) => {
  const [selected, setSelected] = React.useState<MoodOptionType>();
  const [hasAdded, setHasAdded] = React.useState(false);

  const handleAddMood = React.useCallback(() => {
    if (selected) {
      setHasAdded(true);
      onAddMood(selected);
      setSelected(undefined);
    }
  }, [onAddMood, selected]);

  if (hasAdded) {
    return (
      <View style={styles.container}>
        <Text style={styles.title}>Thank you!</Text>
        <Pressable style={styles.button} onPress={() => setHasAdded(false)}>
          <Text style={styles.buttonText}>Add another</Text>
        </Pressable>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <Text style={styles.title}>How are you right now?</Text>
      <View style={styles.emojiList}>
        {moodOptions.map((option) => (
          <View key={option.emoji}>
            <Pressable
              onPress={() => setSelected(option)}
              style={[
                styles.moodItem,
                option.emoji === selected?.emoji ? styles.selected : undefined,
              ]}
            >
              <Text style={styles.emojiText}>{option.emoji}</Text>
            </Pressable>
            <Text style={styles.descriptionText}>
              {option.emoji === selected?.emoji
                ? option.description
                : undefined}
            </Text>
          </View>
        ))}
      </View>
      <Pressable style={styles.button} onPress={handleAddMood}>
        <Text style={styles.buttonText}>Choose</Text>
      </Pressable>
    </View>
  );
};

const styles = StyleSheet.create({
  emojiText: {
    fontSize: 24,
  },
  container: {
    backgroundColor: "rgba(0,0,0,0.2)",
    margin: 10,
    borderRadius: 10,
    padding: 20,
  },
  emojiList: {
    flexDirection: "row",
    justifyContent: "space-between",
    marginBottom: 20,
  },
  moodItem: {
    width: 50,
    height: 50,
    justifyContent: "center",
    alignItems: "center",
  },
  selected: {
    borderWidth: 2,
    backgroundColor: "rgba(0,0,0,0.6)",
    borderColor: "#fff",
    borderRadius: 25,
  },
  descriptionText: {
    color: "#000",
    fontWeight: "bold",
    fontSize: 10,
    textAlign: "center",
  },
  title: {
    textAlign: "center",
    marginBottom: 20,
    fontWeight: "bold",
    fontSize: 18,
  },
  button: {
    alignSelf: "center",
    borderRadius: 100,
    backgroundColor: "rgba(0,0,0,0.6)",
    padding: 10,
    minWidth: 150,
  },
  buttonText: {
    color: "#fff",
    fontWeight: "bold",
    textAlign: "center",
  },
});
```
